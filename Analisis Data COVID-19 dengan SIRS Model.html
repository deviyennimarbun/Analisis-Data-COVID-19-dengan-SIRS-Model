import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import odeint

# Membaca data dari file Excel
file_path = 'data covid19 110724.xls'

# Membaca data dari sheet Oktober2021
data_oktober = pd.read_excel(file_path, sheet_name='Oktober 2021', usecols='B:E', skiprows=5, nrows=31)
data_oktober.columns = ['Tanggal', 'Covid (+) Sembuh', 'Covid (+) dirawat', 'Covid (+) Meninggal']

# Membaca data dari sheet November2021
data_november = pd.read_excel(file_path, sheet_name='November 2021', usecols='B:E', skiprows=5, nrows=30)
data_november.columns = ['Tanggal', 'Covid (+) Sembuh', 'Covid (+) dirawat', 'Covid (+) Meninggal']

# Membaca data dari sheet Desember2021
data_desember = pd.read_excel(file_path, sheet_name='Desember 2021', usecols='B:E', skiprows=5, nrows=31)
data_desember.columns = ['Tanggal', 'Covid (+) Sembuh', 'Covid (+) dirawat', 'Covid (+) Meninggal']

# Menggabungkan semua data
data = pd.concat([data_oktober, data_november, data_desember], ignore_index=True)

# Menambahkan kolom untuk jumlah total kasus Covid (+)
data['Covid (+) Total'] = data['Covid (+) Sembuh'] + data['Covid (+) dirawat'] + data['Covid (+) Meninggal']

# Menentukan jumlah populasi
N = 1000000  # Anggap populasi adalah 1 juta orang

# Mendefinisikan nilai awal
I0 = data['Covid (+) dirawat'].iloc[0]  # Individu infektif awal
R0 = data['Covid (+) Sembuh'].iloc[0] + data['Covid (+) Meninggal'].iloc[0]  # Individu recovered awal
S0 = N - I0 - R0  # Individu susceptible awal

# Waktu simulasi berdasarkan jumlah hari pada data
t = np.linspace(0, len(data)-1, len(data))  # jumlah hari

# Model SI
def SI_model(y, t, N, beta):
    S, I = y
    dSdt = -beta * S * I
    dIdt = beta * S * I
    return dSdt, dIdt

# Kondisi awal untuk model SI
y0_SI = S0, I0

# Solusi model SI
sol_SI = odeint(SI_model, y0_SI, t, args=(N, 0.0005))
S_SI, I_SI = sol_SI.T

# Model SIR
def SIR_model(y, t, N, beta, gamma):
    S, I, R = y
    dSdt = -beta * S * I
    dIdt = beta * S * I - gamma * I
    dRdt = gamma * I
    return dSdt, dIdt, dRdt

# Kondisi awal untuk model SIR
y0_SIR = S0, I0, R0

# Solusi model SIR
sol_SIR = odeint(SIR_model, y0_SIR, t, args=(N, 0.0005, 0.2))
S_SIR, I_SIR, R_SIR = sol_SIR.T
